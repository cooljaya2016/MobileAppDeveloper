  public static int maximum(var [] arr)
  {
      var count = 0;
      var max = 0;
      for(var i = 0; i<arr.length; i++)
      {
          if (arr[i] == 0)
          count = 0;
          else
            count ++;
            max = count > max ? count : max;
      }
     return max;
  }
